
situation_warp_in_progress = {
	picture = GFX_evt_ship_travel
	category = positive

	complete_icon_frame = GFX_situation_outcome_frame_green
	complete_icon = GFX_situation_outcome_positive

	on_start = {
		# monthly rate = 100 / total time * 30
		set_variable = {
			which = total_days
			value = target.warp_time_remaining
		}
		set_variable = {
			which = progress_rate
			value = 100
		}
		divide_variable = {
			which = progress_rate
			value = total_days
		}
	}
	on_progress_complete = {
		custom_tooltip = warp_plugin_arrived
		hidden_effect = {
			# abort_situation = this
			set_situation_locked = yes
		}
	}
	on_abort = {
		target = {
			fleet_event = {
				id = warp_plugin.120
			}
		}
	}
	abort_trigger = {
		target = {
			not = { is_variable_set = warp_time_remaining }
		}
	}
	
	progress_direction = monodirectional

	stages = {
		warp_accelerating = {
			icon = GFX_situation_stage_1
			icon_background = GFX_situation_stage_frame_green
			end = 25
		}
		warp_cruising = {
			icon = GFX_situation_stage_2
			icon_background = GFX_situation_stage_frame_green
			end = 65
		}
		warp_decelerating = {
			icon = GFX_situation_stage_3
			icon_background = GFX_situation_stage_frame_green
			end = 90
		}
		warp_final_approach = {
			icon = GFX_situation_stage_4
			icon_background = GFX_situation_stage_frame_green
			end = 100
		}
	}

	monthly_progress = {
		base = 0
	}

	approach = {
		name = approach_normal_operations
		icon = GFX_situation_approach_fleet
		icon_background = GFX_situation_approach_bg_green
		default = yes
		on_select = {
			custom_tooltip = approach_normal_operations_on_select
			hidden_effect = {
				set_variable = {
					which = progress_rate
					value = 100
				}
				divide_variable = {
					which = progress_rate
					value = total_days
				}
				if = {
					limit = {
						not = { current_stage = warp_final_approach }
						has_situation_flag = warp_in_progress_accelerating
					}
					remove_situation_flag = warp_in_progress_accelerating
					target = {
						multiply_variable = {
							which = warp_time_remaining
							value = 5
						}
						divide_variable = {
							which = warp_time_remaining
							value = 4
						}
						ceiling_variable = warp_time_remaining
					}
				}
			}
		}
	}

	approach = {
		name = approach_accelerate_ftl
		icon = GFX_situation_approach_fleet
		icon_background = GFX_situation_approach_bg_red
		on_select = {
			custom_tooltip = approach_accelerate_ftl_on_select
			hidden_effect = {
				set_variable = {
					which = progress_rate
					value = 125
				}
				divide_variable = {
					which = progress_rate
					value = total_days
				}
				if = {
					limit = {
						not = { has_situation_flag = warp_in_progress_accelerating }
					}
					set_situation_flag = warp_in_progress_accelerating
					target = {
						multiply_variable = {
							which = warp_time_remaining
							value = 4
						}
						divide_variable = {
							which = warp_time_remaining
							value = 5
						}
						ceiling_variable = warp_time_remaining
					}
				}
			}
		}
		allow = {
			custom_tooltip_fail = {
				text = warp_plugin_arrival_soon
				not = { current_stage = warp_final_approach }
			}
		}
		ai_weight = {
			base = 0
		}
	}

	on_monthly = {
	}
}
