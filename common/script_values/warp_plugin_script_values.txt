
warp_travel_time = {
	modifier = {
		set = $DISTANCE$
		is_variable_set = $DISTANCE$
	}
	modifier = {
		divide = $SPEED$
		is_variable_set = $SPEED$
	}
	ceiling = yes
	min = @min_warp_time
}

warp_mia_return_time = {
	modifier = {
		set = $TOTAL$
		is_variable_set = $TOTAL$
	}
	modifier = {
		subtract = $REMAINING$
		is_variable_set = $REMAINING$
	}
	ceiling = yes
	min = @min_warp_time
}
